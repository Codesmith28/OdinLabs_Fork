Java

        
    /**
    * Definition for singly-linked list.
    * public class ListNode {
    * int val;
    * ListNode next;
    * ListNode() {}
    * ListNode(int val) { this.val = val; }
    * ListNode(int val, ListNode next) { this.val = val; this.next = next; }
    * }
    */
    // Using the ListNode definition
    ListNode node2 = new ListNode(20); // create node with value 20
    ListNode node1 = new ListNode(10, node2); // create node with value 10, next points to node2

    // Accessing values
    int val1 = node1.val; // 10
    ListNode next1 = node1.next; // node2
    int val2 = next1.val; // 20

C++

        
    /**
    * Definition for singly-linked list.
    * struct ListNode {
    * int val;
    * ListNode *next;
    * ListNode() : val(0), next(nullptr) {}
    * ListNode(int x) : val(x), next(nullptr) {}
    * ListNode(int x, ListNode *next) : val(x), next(next) {}
    * };
    */
    // Using the ListNode definition
    ListNode* node2 = new ListNode(20); // create node with value 20
    ListNode* node1 = new ListNode(10, node2); // create node with value 10, next points to node2

    // Accessing values
    int val1 = node1->val; // 10
    ListNode* next1 = node1->next; // node2
    int val2 = next1->val; // 20

Java

        
    // Using the Stack class from java.util
    Stack<Integer> stack = new Stack<>();
    stack.push(10); // push 10
    stack.push(20); // push 20
    int top = stack.peek(); // 20
    int popped = stack.pop(); // 20
    boolean isEmpty = stack.isEmpty(); // false
    int size = stack.size(); // 1

C++

        
    // Using the stack container from <stack>
    #include <stack>
    using namespace std;
    stack<int> s;
    s.push(10); // push 10
    s.push(20); // push 20
    int top = s.top(); // 20
    s.pop(); // removes 20
    bool isEmpty = s.empty(); // false
    int size = s.size(); // 1
